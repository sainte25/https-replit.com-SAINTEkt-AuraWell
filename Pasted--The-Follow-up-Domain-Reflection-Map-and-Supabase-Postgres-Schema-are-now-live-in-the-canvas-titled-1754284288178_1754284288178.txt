### ðŸ§± Supabase/Postgres Schema (dbdiagram-style pseudocode)

# Table: users
# - id (uuid)
# - name
# - onboarding_date

# Table: reflections
# - id (uuid)
# - user_id (FK -> users.id)
# - domain (e.g. 'Housing', 'Health')
# - reflection_text (text)
# - tone (nullable string)
# - created_at (timestamp)
# - sccs_points_awarded (integer)
# - action_step (nullable string)

# Table: goals
# - id (uuid)
# - user_id (FK -> users.id)
# - goal_title (text)
# - domain (nullable string)
# - created_at (timestamp)
# - active (boolean)

# Table: sccs_log
# - id (uuid)
# - user_id (FK -> users.id)
# - category (text)
# - description (text)
# - points (integer)
# - source (e.g. 'SIAni session', 'manual entry', 'referral completed')
# - timestamp

# Table: life_events
# - id (uuid)
# - user_id (FK)
# - event_type (e.g. 'shelter move', 'missed PO appt', 'emotional flag')
# - notes (text)
# - created_at

# Table: ai_generated_summaries
# - id (uuid)
# - user_id (FK)
# - reflection_ids (array of FK -> reflections.id)
# - summary_text (text)
# - created_at (timestamp)

# Table: rewards
# - id (uuid)
# - user_id (FK)
# - type (e.g. 'insight', 'encouragement', 'SIAni message')
# - related_sccs_log (FK)
# - message (text or json)
# - delivered_at (timestamp)
